import 'package:core/core.dart';

class CartView extends StatelessWidget {
  const CartView({super.key});

  @override
  Widget build(BuildContext context) {
    return Consumer<PosViewModel>(
      builder: (context, viewModel, child) {
        return Stack(
          children: [
            Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                // 1. Customer Selection Card
                _buildCustomerCard(context, viewModel),
                const SizedBox(height: 24),

                // 2. Cart Items List
                ...viewModel.cart
                    .map((item) => CartItemTile(item: item))
                    .toList(),
                const SizedBox(height: 16),

                // 3. General Order Note
                _buildOrderNote(viewModel),
                const SizedBox(
                    height: 50), // Padding bawah agar tidak tertutup footer
              ],
            ),

            // 4. Customer Search/Add Overlay (Hanya muncul jika state di ViewModel aktif)
            if (viewModel.showCustomerSearch)
              CustomerSearchOverlay(viewModel: viewModel),
          ],
        );
      },
    );
  }

  // --- Component Builders ---

  Widget _buildCustomerCard(BuildContext context, PosViewModel viewModel) {
    const sbBlue = Color(0xFF3B82F6);
    final customer = viewModel.selectedCustomer;

    Color _getLevelColor(String level) {
      switch (level) {
        case 'Gold':
          return Colors.yellow[600]!;
        case 'Silver':
          return Colors.grey[400]!;
        default:
          return Colors.orange[700]!;
      }
    }

    return InkWell(
      onTap: viewModel.toggleCustomerSearch,
      borderRadius: BorderRadius.circular(16.0),
      child: Container(
        padding: const EdgeInsets.all(16.0),
        decoration: BoxDecoration(
          color: customer != null ? Colors.blue[50] : Colors.white,
          borderRadius: BorderRadius.circular(16.0),
          border: Border.all(
            color: customer != null ? sbBlue : Colors.grey[300]!,
            style: customer != null ? BorderStyle.solid : BorderStyle.dashed,
          ),
        ),
        child: customer != null
            ? Stack(
                children: [
                  Row(
                    children: [
                      // Avatar
                      Container(
                        width: 40,
                        height: 40,
                        decoration: BoxDecoration(
                          shape: BoxShape.circle,
                          color: _getLevelColor(customer.level),
                          boxShadow: [
                            BoxShadow(
                                color: Colors.black.withOpacity(0.1),
                                blurRadius: 3),
                          ],
                        ),
                        child: Center(
                          child: Text(
                            customer.name.substring(0, 1),
                            style: const TextStyle(
                                color: Colors.white,
                                fontWeight: FontWeight.bold,
                                fontSize: 18),
                          ),
                        ),
                      ),
                      const SizedBox(width: 12),
                      // Info
                      Expanded(
                        child: Column(
                          crossAxisAlignment: CrossAxisAlignment.start,
                          children: [
                            Row(
                              children: [
                                Text(
                                  customer.name,
                                  style: const TextStyle(
                                      fontWeight: FontWeight.bold,
                                      fontSize: 16,
                                      color: Color(0xFF333333)),
                                ),
                                if (customer.level == 'Gold')
                                  Padding(
                                    padding: const EdgeInsets.only(left: 4.0),
                                    child: Icon(LucideIcons.crown,
                                        size: 16,
                                        color: _getLevelColor(customer.level),
                                        fill: _getLevelColor(customer.level)),
                                  ),
                              ],
                            ),
                            const SizedBox(height: 2),
                            Row(
                              children: [
                                Text('${customer.points} Poin',
                                    style: const TextStyle(
                                        color: sbBlue,
                                        fontWeight: FontWeight.bold,
                                        fontSize: 12)),
                                const SizedBox(width: 8),
                                Text('â€¢ ${customer.phone}',
                                    style: TextStyle(
                                        color: Colors.grey[500], fontSize: 12)),
                              ],
                            ),
                          ],
                        ),
                      ),
                      // Remove Button
                      InkWell(
                        onTap: () => viewModel.selectCustomer(null),
                        child: const Padding(
                          padding: EdgeInsets.only(left: 8.0),
                          child:
                              Icon(LucideIcons.x, size: 18, color: Colors.grey),
                        ),
                      )
                    ],
                  ),
                  // Notes if available
                  if (customer.notes != null && customer.notes!.isNotEmpty)
                    Positioned(
                      top: 0,
                      right: 0,
                      child: Container(
                        padding: const EdgeInsets.symmetric(
                            horizontal: 8.0, vertical: 4.0),
                        decoration: BoxDecoration(
                          color: Colors.yellow[100],
                          borderRadius: const BorderRadius.only(
                              bottomLeft: Radius.circular(8.0)),
                        ),
                        child: Text(
                          'Catatan: ${customer.notes}',
                          style: TextStyle(
                              color: Colors.yellow[800],
                              fontSize: 10,
                              fontWeight: FontWeight.w500),
                        ),
                      ),
                    ),
                ],
              )
            : Row(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(LucideIcons.user, size: 20, color: Colors.grey[500]),
                  const SizedBox(width: 8),
                  Text(
                    'Pilih Pelanggan / Member',
                    style: TextStyle(
                        fontWeight: FontWeight.w500,
                        color: Colors.grey[500],
                        fontSize: 14),
                  ),
                ],
              ),
      ),
    );
  }

  Widget _buildOrderNote(PosViewModel viewModel) {
    const sbYellow = Color(0xFFFBBF24); // Representing yellow-700

    return Container(
      padding: const EdgeInsets.all(12.0),
      decoration: BoxDecoration(
        color: Colors.yellow[50],
        borderRadius: BorderRadius.circular(16.0),
        border: Border.all(color: Colors.yellow[100]!),
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Row(
            children: [
              Icon(LucideIcons.fileText, size: 16, color: sbYellow),
              const SizedBox(width: 8),
              Text(
                'CATATAN PESANAN',
                style: TextStyle(
                    color: Colors.yellow[800],
                    fontSize: 12,
                    fontWeight: FontWeight.bold,
                    letterSpacing: 0.5),
              ),
            ],
          ),
          const SizedBox(height: 8),
          TextFormField(
            initialValue: viewModel.orderNote,
            onChanged: viewModel.setOrderNote,
            maxLines: 3,
            minLines: 2,
            decoration: InputDecoration(
              hintText: "Contoh: Bungkus dipisah, Meja nomor 5...",
              hintStyle: TextStyle(color: Colors.grey[300]),
              filled: true,
              fillColor: Colors.white,
              contentPadding: const EdgeInsets.all(8.0),
              border: OutlineInputBorder(
                borderRadius: BorderRadius.circular(8.0),
                borderSide: BorderSide(color: Colors.grey[200]!),
              ),
              enabledBorder: OutlineInputBorder(
                borderRadius: BorderRadius.circular(8.0),
                borderSide: BorderSide(color: Colors.grey[200]!),
              ),
              focusedBorder: const OutlineInputBorder(
                borderRadius: BorderRadius.circular(8.0),
                borderSide: BorderSide(color: sbBlue, width: 1),
              ),
            ),
            style: const TextStyle(fontSize: 14),
          ),
        ],
      ),
    );
  }
}

class CustomerSearchOverlay extends StatelessWidget {
  final PosViewModel viewModel;

  const CustomerSearchOverlay({super.key, required this.viewModel});

  @override
  Widget build(BuildContext context) {
    // Implementasi ini adalah simulasi overlay di dalam CartSheet.
    // Dalam aplikasi nyata, ini bisa menjadi BottomSheet baru atau navigasi.
    // Untuk tujuan konversi, kita akan membuatnya menumpuk di atas konten CartView.

    const sbBlue = Color(0xFF3B82F6);

    return Positioned.fill(
      child: Container(
        color: Colors.white, // Menutupi CartView di belakang
        child: Column(
          children: [
            // Header Search
            Padding(
              padding: const EdgeInsets.only(
                  left: 8.0, top: 8.0, right: 16.0, bottom: 8.0),
              child: Row(
                children: [
                  IconButton(
                    icon:
                        Icon(LucideIcons.chevronLeft, color: Colors.grey[600]),
                    onPressed: viewModel.toggleCustomerSearch,
                  ),
                  Text(
                    viewModel.isAddingCustomer
                        ? 'Pelanggan Baru'
                        : 'Cari Pelanggan',
                    style: const TextStyle(
                        fontWeight: FontWeight.bold, fontSize: 18),
                  ),
                ],
              ),
            ),
            const Divider(height: 1),

            // Form / Search Content
            Expanded(
              child: SingleChildScrollView(
                padding: const EdgeInsets.all(16.0),
                child: viewModel.isAddingCustomer
                    ? _buildAddCustomerForm(viewModel, sbBlue)
                    : _buildCustomerSearchList(viewModel, sbBlue),
              ),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildAddCustomerForm(PosViewModel viewModel, Color sbBlue) {
    // Implementasi Form Pelanggan Baru
    return Column(
      children: [
        const SizedBox(height: 16),
        Container(
          width: 80,
          height: 80,
          decoration: BoxDecoration(
            shape: BoxShape.circle,
            color: Colors.blue[50],
            border:
                Border.all(color: sbBlue, style: BorderStyle.dashed, width: 2),
          ),
          child: const Icon(LucideIcons.userPlus, size: 32, color: sbBlue),
        ),
        const SizedBox(height: 24),
        _buildInputField(
          label: 'Nama Pelanggan *',
          icon: LucideIcons.user,
          placeholder: 'Nama lengkap',
          value: viewModel.newCustomerForm.name,
          onChanged: (v) => viewModel.updateNewCustomerForm('name', v),
        ),
        _buildInputField(
          label: 'Nomor Telepon *',
          icon: LucideIcons.phone,
          placeholder: '08xxxxxxxx',
          value: viewModel.newCustomerForm.phone,
          onChanged: (v) => viewModel.updateNewCustomerForm('phone', v),
          keyboardType: TextInputType.phone,
        ),
        _buildInputField(
          label: 'Email (Opsional)',
          icon: LucideIcons.mail,
          placeholder: 'email@example.com',
          value: viewModel.newCustomerForm.email,
          onChanged: (v) => viewModel.updateNewCustomerForm('email', v),
          keyboardType: TextInputType.emailAddress,
        ),
        const SizedBox(height: 8),
        _buildNotesField(
          label: 'Catatan Khusus',
          placeholder: 'Contoh: Alergi kacang...',
          value: viewModel.newCustomerForm.notes,
          onChanged: (v) => viewModel.updateNewCustomerForm('notes', v),
        ),
        const SizedBox(height: 24),
        SizedBox(
          width: double.infinity,
          child: ElevatedButton.icon(
            onPressed: viewModel.handleSaveNewCustomer,
            icon: const Icon(LucideIcons.save, size: 18, color: Colors.white),
            label: const Text('Simpan & Pilih Pelanggan',
                style: TextStyle(
                    color: Colors.white,
                    fontWeight: FontWeight.bold,
                    fontSize: 16)),
            style: ElevatedButton.styleFrom(
              backgroundColor: sbBlue,
              padding: const EdgeInsets.symmetric(vertical: 16),
              shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(12)),
              elevation: 5,
            ),
          ),
        ),
      ],
    );
  }

  Widget _buildCustomerSearchList(PosViewModel viewModel, Color sbBlue) {
    // Implementasi Pencarian Pelanggan dan Daftar Hasil
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        TextField(
          onChanged: viewModel.setCustomerQuery,
          autofocus: true,
          decoration: InputDecoration(
            hintText: 'Ketik Nama atau Nomor HP...',
            prefixIcon: const Icon(LucideIcons.search, color: Colors.grey),
            filled: true,
            fillColor: Colors.grey[50],
            border: OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: Colors.grey[200]!),
            ),
            enabledBorder: OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: Colors.grey[200]!),
            ),
            focusedBorder: OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: sbBlue, width: 1.5),
            ),
            contentPadding: const EdgeInsets.symmetric(vertical: 16.0),
          ),
          style: const TextStyle(fontSize: 16),
        ),
        const SizedBox(height: 24),

        const Text('HASIL PENCARIAN',
            style: TextStyle(
                color: Colors.grey, fontWeight: FontWeight.bold, fontSize: 12)),
        const SizedBox(height: 12),

        // Guest Option
        _buildCustomerTile(
          name: 'Guest (Tamu)',
          phone: 'Tanpa data pelanggan',
          level: '',
          onTap: () => viewModel.selectCustomer(null),
          avatarColor: Colors.grey[200]!,
          avatarIcon: Icon(LucideIcons.user, size: 20, color: Colors.grey[500]),
        ),
        const SizedBox(height: 8),

        // Filtered Results
        ...viewModel.filteredCustomers
            .map((c) => Padding(
                  padding: const EdgeInsets.only(bottom: 8.0),
                  child: _buildCustomerTile(
                    customer: c,
                    onTap: () => viewModel.selectCustomer(c),
                    avatarColor: _getLevelColor(c.level),
                    avatarText: c.name.substring(0, 1),
                  ),
                ))
            .toList(),

        const SizedBox(height: 24),
        // Add New Customer Button
        SizedBox(
          width: double.infinity,
          child: OutlinedButton.icon(
            onPressed: viewModel.toggleAddCustomerForm,
            icon: const Icon(LucideIcons.userPlus, size: 18, color: sbBlue),
            label: const Text('+ Tambah Pelanggan Baru',
                style: TextStyle(color: sbBlue, fontWeight: FontWeight.bold)),
            style: OutlinedButton.styleFrom(
              padding: const EdgeInsets.symmetric(vertical: 16),
              side: const BorderSide(
                  color: sbBlue, width: 2, style: BorderStyle.dashed),
              shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(12)),
            ),
          ),
        ),
      ],
    );
  }

  Color _getLevelColor(String level) {
    switch (level) {
      case 'Gold':
        return Colors.yellow[500]!;
      case 'Silver':
        return Colors.grey[400]!;
      default:
        return Colors.orange[700]!;
    }
  }

  Widget _buildCustomerTile({
    Customer? customer,
    required VoidCallback onTap,
    required Color avatarColor,
    String? name,
    String? phone,
    String? level,
    String? avatarText,
    Widget? avatarIcon,
  }) {
    name ??= customer!.name;
    phone ??= customer!.phone;
    level ??= customer!.level;
    final points = customer?.points ?? 0;

    return InkWell(
      onTap: onTap,
      borderRadius: BorderRadius.circular(12.0),
      child: Container(
        padding: const EdgeInsets.all(16.0),
        decoration: BoxDecoration(
          color: Colors.white,
          borderRadius: BorderRadius.circular(12.0),
          border: Border.all(color: Colors.grey[100]!),
          boxShadow: [
            BoxShadow(color: Colors.grey.withOpacity(0.1), blurRadius: 4)
          ],
        ),
        child: Row(
          children: [
            // Avatar
            Container(
              width: 40,
              height: 40,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                color: avatarColor,
              ),
              child: Center(
                child: avatarIcon ??
                    Text(
                      avatarText!,
                      style: const TextStyle(
                          color: Colors.white,
                          fontWeight: FontWeight.bold,
                          fontSize: 16),
                    ),
              ),
            ),
            const SizedBox(width: 12),
            // Info
            Expanded(
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  Row(
                    children: [
                      Text(name,
                          style: const TextStyle(
                              fontWeight: FontWeight.bold, fontSize: 16)),
                      if (level == 'Gold')
                        Padding(
                          padding: const EdgeInsets.only(left: 4.0),
                          child: Icon(LucideIcons.crown,
                              size: 14,
                              color: _getLevelColor(level),
                              fill: _getLevelColor(level)),
                        ),
                    ],
                  ),
                  Text(phone,
                      style: TextStyle(color: Colors.grey[500], fontSize: 12)),
                ],
              ),
            ),
            // Points/Level
            if (customer != null)
              Column(
                crossAxisAlignment: CrossAxisAlignment.end,
                children: [
                  Row(
                    children: [
                      Icon(LucideIcons.star,
                          size: 14,
                          color: Color(0xFF3B82F6),
                          fill: Color(0xFF3B82F6)),
                      const SizedBox(width: 4),
                      Text('$points',
                          style: const TextStyle(
                              color: Color(0xFF3B82F6),
                              fontWeight: FontWeight.bold)),
                    ],
                  ),
                  Text('$level Member',
                      style: TextStyle(color: Colors.grey[400], fontSize: 10)),
                ],
              ),
          ],
        ),
      ),
    );
  }

  Widget _buildInputField({
    required String label,
    required IconData icon,
    required String placeholder,
    required String value,
    required ValueChanged<String> onChanged,
    TextInputType keyboardType = TextInputType.text,
  }) {
    return Padding(
      padding: const EdgeInsets.only(bottom: 16.0),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Text(label,
              style: TextStyle(
                  color: Colors.grey[500],
                  fontWeight: FontWeight.bold,
                  fontSize: 12)),
          const SizedBox(height: 4),
          TextFormField(
            initialValue: value,
            onChanged: onChanged,
            keyboardType: keyboardType,
            decoration: InputDecoration(
              hintText: placeholder,
              prefixIcon: Icon(icon, size: 18, color: Colors.grey[400]),
              filled: true,
              fillColor: Colors.grey[50],
              contentPadding:
                  const EdgeInsets.symmetric(vertical: 12.0, horizontal: 10.0),
              border: OutlineInputBorder(
                borderRadius: BorderRadius.circular(12.0),
                borderSide: BorderSide(color: Colors.grey[200]!),
              ),
              enabledBorder: OutlineInputBorder(
                borderRadius: BorderRadius.circular(12.0),
                borderSide: BorderSide(color: Colors.grey[200]!),
              ),
              focusedBorder: const OutlineInputBorder(
                borderRadius: BorderRadius.circular(12.0),
                borderSide: BorderSide(color: Color(0xFF3B82F6), width: 1.5),
              ),
            ),
            style: const TextStyle(fontSize: 14),
          ),
        ],
      ),
    );
  }

  Widget _buildNotesField({
    required String label,
    required String placeholder,
    required String value,
    required ValueChanged<String> onChanged,
  }) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Text(label,
            style: TextStyle(
                color: Colors.grey[500],
                fontWeight: FontWeight.bold,
                fontSize: 12)),
        const SizedBox(height: 4),
        TextFormField(
          initialValue: value,
          onChanged: onChanged,
          maxLines: 4,
          minLines: 3,
          decoration: InputDecoration(
            hintText: placeholder,
            filled: true,
            fillColor: Colors.grey[50],
            contentPadding: const EdgeInsets.all(12.0),
            border: OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: Colors.grey[200]!),
            ),
            enabledBorder: OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: Colors.grey[200]!),
            ),
            focusedBorder: const OutlineInputBorder(
              borderRadius: BorderRadius.circular(12.0),
              borderSide: BorderSide(color: Color(0xFF3B82F6), width: 1.5),
            ),
          ),
          style: const TextStyle(fontSize: 14),
        ),
      ],
    );
  }
}
